let kioskHoldStart = 0; let activeFingers = 0; let holdTimer = null; function resetHold(){ kioskHoldStart = 0; if(holdTimer) clearTimeout(holdTimer); holdTimer = null; } document.addEventListener("touchstart", function(e){ if(!document.body.classList.contains("kiosk")) return; activeFingers = e.touches.length; if(activeFingers === 3){ kioskHoldStart = Date.now(); holdTimer = setTimeout(()=>{ // Sicherheitsreset resetHold(); }, 2000); } }); document.addEventListener("touchmove", function(e){ // Wenn Fingerzahl sich ändert → abbrechen if(e.touches.length !== 3){ resetHold(); } }); document.addEventListener("touchend", function(e){ if(!document.body.classList.contains("kiosk")) return; if(activeFingers === 3){ let duration = Date.now() - kioskHoldStart; // 🔑 1.2 Sekunden halten if(duration > 1200){ exitKiosk(); } } resetHold(); });
